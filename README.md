# Simple bluetooth library for android
# installation

Add it to your root build.gradle

'''
	allprojects {
		repositories {
			...
			maven { url 'https://jitpack.io' }
		}
	}
  '''
  
 then add the dependency
 
'''
 	dependencies {
	        compile 'com.github.anouarkappitou:BluetoothLib-android:v1.0'
	}
'''
